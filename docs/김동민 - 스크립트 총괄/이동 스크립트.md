# 이동 스크립트

달리기 : 걷기 중 Shift

앉기 : 서 있는 상태에서 ctrl

슬라이딩 : 달리기 중 ctrl

벽타기 : 벽에 붙어서 space

위치 : **StarterPlayer** → **StarterCharacterScripts**

파일 명 SlideScript

```jsx
local UserInputService = game:GetService("UserInputService")
local RunService       = game:GetService("RunService")
local Players          = game:GetService("Players")
local TweenService     = game:GetService("TweenService")

local Character = script.Parent
local Humanoid  = Character:WaitForChild("Humanoid")
local Animator  = Humanoid:WaitForChild("Animator")
local RootPart  = Character:WaitForChild("HumanoidRootPart")
local Player    = Players.LocalPlayer

local defaultHipHeight = Humanoid.HipHeight 

local Config = {
	WalkSpeed       = 16,
	RunSpeed        = 28,
	CrouchSpeed     = 8,
	StandHipHeight  = defaultHipHeight,
	CrouchHipHeight = 0.1,
	StandCamOffset  = Vector3.new(0,  0,   0),
	CrouchCamOffset = Vector3.new(0, -1.5, 0),
	SlideForce      = 100,
	SlideDuration   = 0.6,
	SlideCooldown   = 0.5,
	SlideJumpBoostDuration = 0.2,

	-- ★ 벽타기 설정
	WallDetectRange       = 2.0,   -- 벽 감지 거리
	WallClimbSpeed        = 10,    -- W키 오르기 속도
	WallSlideSpeed        = -3,    -- 아무것도 안 누를 때 흘러내림 속도
	WallStickForce        = 8,     -- 벽에 붙어있는 힘 (수평)
	WallClimbMaxTime      = 3.0,   -- 최대 벽타기 유지 시간
	WallJumpHForce        = 55,    -- 벽 점프 수평 속도
	WallJumpVForce        = 45,    -- 벽 점프 수직 속도
	WallJumpBoostDuration = 0.18,  -- 벽 점프 부스트 지속 시간
	WallJumpCooldownTime  = 0.5,   -- 동일 벽 재부착 방지 시간

	Anims = {
		Crouch    = "rbxassetid://124107764894700",
		Slide     = "rbxassetid://124107764894700",
		WallClimb = "rbxassetid://124107764894700", -- ← 교체 필요
	}
}

-- ══════════════════════════════════════════
--  상태 정의
-- ══════════════════════════════════════════
local State = {
	WALKING       = "Walking",
	RUNNING       = "Running",
	CROUCHING     = "Crouching",
	SLIDING       = "Sliding",
	WALL_CLIMBING = "WallClimb",  
}

local currentState  = State.WALKING
local slideDebounce = false
local slideActive   = false
local slideContext  = { lv = nil, attachment = nil }

-- ★ 벽타기 상태
local wallClimbActive  = false
local wallClimbContext = { lv = nil, attachment = nil, wallNormal = nil }
local wallJumpCooldown = false

-- ══════════════════════════════════════════
--  공통 유틸
-- ══════════════════════════════════════════
local function isGrounded()
	return Humanoid.FloorMaterial ~= Enum.Material.Air
end

local function setCameraOffset(targetOffset)
	TweenService:Create(
		Humanoid,
		TweenInfo.new(0.15, Enum.EasingStyle.Quad, Enum.EasingDirection.Out),
		{ CameraOffset = targetOffset }
	):Play()
end

-- ★ 벽 레이캐스트 파라미터 (캐릭터 자신 제외)
local wallRayParams = RaycastParams.new()
wallRayParams.FilterDescendantsInstances = {Character}
wallRayParams.FilterType = Enum.RaycastFilterType.Exclude

local function getWallInFront(range)
	return workspace:Raycast(
		RootPart.Position,
		RootPart.CFrame.LookVector * (range or Config.WallDetectRange),
		wallRayParams
	)
end

-- Normal의 Y 성분이 작으면 수직 벽
local function isWall(result)
	if not result then return false end
	return math.abs(result.Normal.Y) < 0.25
end

-- ══════════════════════════════════════════
--  Speed GUI
-- ══════════════════════════════════════════
local ScreenGui = Instance.new("ScreenGui")
ScreenGui.Name, ScreenGui.ResetOnSpawn = "SpeedGUI", false
ScreenGui.Parent = Player.PlayerGui

local Frame = Instance.new("Frame")
Frame.Size, Frame.Position = UDim2.new(0, 160, 0, 64), UDim2.new(0, 12, 1, -76)
Frame.BackgroundColor3      = Color3.fromRGB(10, 10, 10)
Frame.BackgroundTransparency = 0.4
Frame.BorderSizePixel       = 0
Frame.Parent                = ScreenGui
Instance.new("UICorner", Frame).CornerRadius = UDim.new(0, 10)

local StateLabel = Instance.new("TextLabel")
StateLabel.Size, StateLabel.Position = UDim2.new(1,-10,0.5,0), UDim2.new(0,8,0,0)
StateLabel.BackgroundTransparency = 1
StateLabel.TextXAlignment = Enum.TextXAlignment.Left
StateLabel.TextScaled, StateLabel.Font = true, Enum.Font.GothamBold
StateLabel.Text   = "▶ Walking"
StateLabel.Parent = Frame

local SpeedLabel = Instance.new("TextLabel")
SpeedLabel.Size, SpeedLabel.Position = UDim2.new(1,-10,0.5,0), UDim2.new(0,8,0.5,0)
SpeedLabel.BackgroundTransparency = 1
SpeedLabel.TextColor3 = Color3.fromRGB(200, 200, 200)
SpeedLabel.TextXAlignment = Enum.TextXAlignment.Left
SpeedLabel.TextScaled, SpeedLabel.Font = true, Enum.Font.Gotham
SpeedLabel.Text   = "Speed: 0 su/s"
SpeedLabel.Parent = Frame

local StateColors = {
	[State.WALKING]       = Color3.fromRGB(200, 200, 200),
	[State.RUNNING]       = Color3.fromRGB(100, 220, 100),
	[State.CROUCHING]     = Color3.fromRGB(100, 180, 255),
	[State.SLIDING]       = Color3.fromRGB(255, 140,  50),
	[State.WALL_CLIMBING] = Color3.fromRGB(255,  80, 200), -- ★ 신규
}

-- ══════════════════════════════════════════
--  애니메이션 로드
-- ══════════════════════════════════════════
local function loadAnim(id)
	local a = Instance.new("Animation")
	a.AnimationId = id
	return Animator:LoadAnimation(a)
end

local Tracks = {
	Crouch    = loadAnim(Config.Anims.Crouch),
	Slide     = loadAnim(Config.Anims.Slide),
	WallClimb = loadAnim(Config.Anims.WallClimb),
}

local function stopAllTracks()
	for _, track in pairs(Tracks) do track:Stop() end
end

-- ══════════════════════════════════════════
--  상태 전환
-- ══════════════════════════════════════════
local function setState(newState)
	if currentState == newState then return end
	currentState = newState
	stopAllTracks()

	if newState == State.WALKING then
		Humanoid.WalkSpeed  = Config.WalkSpeed
		Humanoid.HipHeight  = Config.StandHipHeight
		Humanoid.AutoRotate = true
		setCameraOffset(Config.StandCamOffset)
	elseif newState == State.RUNNING then
		Humanoid.WalkSpeed  = Config.RunSpeed
		Humanoid.HipHeight  = Config.StandHipHeight
		Humanoid.AutoRotate = true
		setCameraOffset(Config.StandCamOffset)
	elseif newState == State.CROUCHING then
		Humanoid.WalkSpeed  = Config.CrouchSpeed
		Humanoid.HipHeight  = Config.CrouchHipHeight
		Humanoid.AutoRotate = true
		setCameraOffset(Config.CrouchCamOffset)
		Tracks.Crouch:Play()
	end
end

-- ══════════════════════════════════════════
--  슬라이딩
-- ══════════════════════════════════════════
local function startSlide()
	if slideDebounce or currentState ~= State.RUNNING then return end
	if not isGrounded() then return end

	slideDebounce = true
	currentState  = State.SLIDING
	slideActive   = true

	stopAllTracks()
	Tracks.Slide:Play()
	Humanoid.HipHeight = Config.CrouchHipHeight
	Humanoid.WalkSpeed = 0
	setCameraOffset(Config.CrouchCamOffset)

	local attachment = Instance.new("Attachment")
	attachment.Parent = RootPart
	local lv = Instance.new("LinearVelocity")
	lv.Attachment0        = attachment
	local look            = RootPart.CFrame.LookVector
	lv.VectorVelocity     = Vector3.new(look.X, 0, look.Z) * Config.SlideForce
	lv.ForceLimitsEnabled = true
	lv.ForceLimitMode     = Enum.ForceLimitMode.PerAxis
	lv.MaxAxesForce       = Vector3.new(25000, 0, 25000)  -- ★ Y축 힘 차단
	lv.Parent             = RootPart
	slideContext.lv, slideContext.attachment = lv, attachment

	local elapsed = 0
	while elapsed < Config.SlideDuration and slideActive do
		elapsed += RunService.Heartbeat:Wait()
	end

	if slideActive then
		slideActive = false
		slideContext.lv, slideContext.attachment = nil, nil
		Tracks.Slide:Stop()
		lv:Destroy()
		attachment:Destroy()
		if UserInputService:IsKeyDown(Enum.KeyCode.LeftControl) then
			setState(State.CROUCHING)
		elseif UserInputService:IsKeyDown(Enum.KeyCode.LeftShift) then
			setState(State.RUNNING)
		else
			setState(State.WALKING)
		end
	end

	task.wait(Config.SlideCooldown)
	slideDebounce = false
end

local function cancelSlideWithJump()
	local lv  = slideContext.lv
	local att = slideContext.attachment
	if not lv or not att then return end

	local slideDir = Vector3.new(lv.VectorVelocity.X, 0, lv.VectorVelocity.Z)
	slideActive = false
	slideContext.lv, slideContext.attachment = nil, nil

	Tracks.Slide:Stop()
	lv:Destroy()
	att:Destroy()

	currentState = State.WALKING
	Humanoid.HipHeight = Config.StandHipHeight
	Humanoid.WalkSpeed = Config.WalkSpeed
	setCameraOffset(Config.StandCamOffset)
	Humanoid.Jump = true

	local boostAtt = Instance.new("Attachment")
	boostAtt.Parent = RootPart
	local boostLV = Instance.new("LinearVelocity")
	boostLV.Attachment0        = boostAtt
	boostLV.VectorVelocity     = slideDir
	boostLV.ForceLimitsEnabled = true
	boostLV.ForceLimitMode     = Enum.ForceLimitMode.PerAxis
	boostLV.MaxAxesForce       = Vector3.new(18000, 0, 18000)  -- ★ Y축 힘 차단
	boostLV.Parent             = RootPart
	task.delay(Config.SlideJumpBoostDuration, function()
		if boostLV and boostLV.Parent then boostLV:Destroy() end
		if boostAtt and boostAtt.Parent then boostAtt:Destroy() end
	end)
end

-- ══════════════════════════════════════════
--  ★ 벽타기
-- ══════════════════════════════════════════
local function cleanupWallClimb()
	if wallClimbContext.lv and wallClimbContext.lv.Parent then
		wallClimbContext.lv:Destroy()
	end
	if wallClimbContext.attachment and wallClimbContext.attachment.Parent then
		wallClimbContext.attachment:Destroy()
	end
	wallClimbContext.lv         = nil
	wallClimbContext.attachment = nil
	wallClimbContext.wallNormal = nil
	Humanoid.AutoRotate = true
end

local function startWallClimb(wallResult)
	if currentState == State.WALL_CLIMBING then return end
	if isGrounded() or wallJumpCooldown then return end

	currentState    = State.WALL_CLIMBING
	wallClimbActive = true

	stopAllTracks()
	Tracks.WallClimb:Play()

	local wallNormal = wallResult.Normal
	wallClimbContext.wallNormal = wallNormal

	-- 캐릭터를 벽 방향으로 회전
	Humanoid.AutoRotate = false
	Humanoid.WalkSpeed  = 0
	RootPart.CFrame     = CFrame.new(RootPart.Position, RootPart.Position - wallNormal)

	local attachment = Instance.new("Attachment")
	attachment.Parent = RootPart

	local lv = Instance.new("LinearVelocity")
	lv.Attachment0    = attachment
	lv.VectorVelocity = Vector3.new(
		-wallNormal.X * Config.WallStickForce,
		0,
		-wallNormal.Z * Config.WallStickForce
	)
	lv.MaxForce = 50000
	lv.Parent   = RootPart

	wallClimbContext.lv         = lv
	wallClimbContext.attachment = attachment

	local elapsed = 0
	while wallClimbActive and elapsed < Config.WallClimbMaxTime do
		elapsed += RunService.Heartbeat:Wait()

		if isGrounded() then break end

		-- 벽이 사라졌는지 확인
		if not isWall(getWallInFront(Config.WallDetectRange)) then break end

		-- S키 → 벽에서 놓기
		if UserInputService:IsKeyDown(Enum.KeyCode.S) then break end

		-- W키 오르기 / 기본 흘러내림
		local climbY = UserInputService:IsKeyDown(Enum.KeyCode.W)
			and Config.WallClimbSpeed
			or  Config.WallSlideSpeed

		lv.VectorVelocity = Vector3.new(
			-wallNormal.X * Config.WallStickForce,
			climbY,
			-wallNormal.Z * Config.WallStickForce
		)
	end

	-- 자연 종료
	wallClimbActive = false
	cleanupWallClimb()
	Tracks.WallClimb:Stop()

	if isGrounded() then
		setState(State.WALKING)
	else
		currentState        = State.WALKING
		Humanoid.WalkSpeed  = Config.WalkSpeed
		Humanoid.HipHeight  = Config.StandHipHeight
		setCameraOffset(Config.StandCamOffset)
	end
end

local function performWallJump()
	local normal = wallClimbContext.wallNormal
	if not normal then return end

	wallClimbActive = false  -- 루프 종료 트리거
	cleanupWallClimb()
	Tracks.WallClimb:Stop()

	currentState        = State.WALKING
	Humanoid.WalkSpeed  = Config.WalkSpeed
	Humanoid.HipHeight  = Config.StandHipHeight
	setCameraOffset(Config.StandCamOffset)

	-- 벽 반대 방향(Normal) + 위로 점프
	local jumpAtt = Instance.new("Attachment")
	jumpAtt.Parent = RootPart
	local jumpLV = Instance.new("LinearVelocity")
	jumpLV.Attachment0    = jumpAtt
	jumpLV.VectorVelocity = normal * Config.WallJumpHForce
		+ Vector3.new(0, Config.WallJumpVForce, 0)
	jumpLV.MaxForce       = 60000
	jumpLV.Parent         = RootPart

	task.delay(Config.WallJumpBoostDuration, function()
		if jumpLV and jumpLV.Parent then jumpLV:Destroy() end
		if jumpAtt and jumpAtt.Parent then jumpAtt:Destroy() end
	end)

	-- 같은 벽 즉시 재부착 방지
	wallJumpCooldown = true
	task.delay(Config.WallJumpCooldownTime, function()
		wallJumpCooldown = false
	end)
end

-- ══════════════════════════════════════════
--  Heartbeat: GUI + 벽 자동 감지
-- ══════════════════════════════════════════
RunService.Heartbeat:Connect(function()
	local vel   = RootPart.Velocity
	local speed = math.floor(Vector3.new(vel.X, 0, vel.Z).Magnitude)
	SpeedLabel.Text       = "Speed: " .. speed .. " su/s"
	StateLabel.Text       = "▶ " .. currentState
	StateLabel.TextColor3 = StateColors[currentState] or Color3.new(1,1,1)

	-- ★ 벽 자동 감지: 공중 + 일반 이동 상태 + 쿨다운 없음
	if not isGrounded()
		and not wallJumpCooldown
		and (currentState == State.WALKING or currentState == State.RUNNING)
	then
		local wallResult = getWallInFront(Config.WallDetectRange)
		if isWall(wallResult) then
			local velFlat  = Vector3.new(vel.X, 0, vel.Z)
			local lookFlat = Vector3.new(RootPart.CFrame.LookVector.X, 0, RootPart.CFrame.LookVector.Z)
			-- 전진 속도가 충분하고 시선 방향이 이동 방향과 일치할 때만 잡기
			if velFlat.Magnitude > 3
				and lookFlat.Magnitude > 0.01
				and lookFlat.Unit:Dot(velFlat.Unit) > 0.5
			then
				task.spawn(startWallClimb, wallResult)
			end
		end
	end
end)

-- ══════════════════════════════════════════
--  입력 처리
-- ══════════════════════════════════════════
UserInputService.InputBegan:Connect(function(input, gameProcessed)
	if gameProcessed then return end

	if input.KeyCode == Enum.KeyCode.LeftShift then
		if currentState == State.WALKING then
			setState(State.RUNNING)
		end

	elseif input.KeyCode == Enum.KeyCode.LeftControl then
		if currentState == State.RUNNING then
			task.spawn(startSlide)
		elseif currentState == State.WALKING then
			setState(State.CROUCHING)
		end

	elseif input.KeyCode == Enum.KeyCode.Space then
		if currentState == State.SLIDING then
			cancelSlideWithJump()
		elseif currentState == State.WALL_CLIMBING then
			performWallJump()       -- ★ 벽 점프
		end
	end
end)

UserInputService.InputEnded:Connect(function(input, gameProcessed)
	if input.KeyCode == Enum.KeyCode.LeftShift then
		if currentState == State.RUNNING then
			setState(State.WALKING)
		end

	elseif input.KeyCode == Enum.KeyCode.LeftControl then
		if currentState == State.CROUCHING then
			if UserInputService:IsKeyDown(Enum.KeyCode.LeftShift) then
				setState(State.RUNNING)
			else
				setState(State.WALKING)
			end
		end
	end
end)
```
